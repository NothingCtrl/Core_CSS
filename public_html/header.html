Groups

Core_CSS 
        <div class="header bg-red-A400">
            <div class="header-container" id="header-main-group" style="display: block">
                <button class="nav-button bg-red-A400"><span></span></button>
                <span class="app-title">Core_CSS</span>
                <button class="tool-button place-right bg-red-A400" id="searchButton"><span class="i-search icon"></span></button>
            </div>
            <div class="header-container" id="header-search-group" style="display: none">
                <button class="tool-button bg-red-A400" id="closeSearchButton"><span class="i-back icon"></span></button>
                <div class="input-control">
                    <input type="text" name="search" class="search-in-list" placeholder="Search...">
                </div>
            </div>
        </div>

        <script>
            $('#searchButton, #closeSearchButton').off('click').on('click', function(){
                var b = $(this);

                if (b.attr('id') == 'searchButton') {
                    $("#header-main-group").hide();
                    $("#header-search-group").show();
                    $(".search-in-list").focus();
                } else {
                    $("#header-main-group").show();
                    $("#header-search-group").hide();
                }
            });
        </script>
