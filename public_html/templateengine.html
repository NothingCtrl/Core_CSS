Template engine

Core_CSS contains micro template engine. This feature specified in function coreTemplate or $.Template. The function takes 2 parameters: template as string and vars defined as object. 

        var template =
         'My skills:' +
         '<%if(this.showSkills) {%>' +
         '<ul>' +
         '<%for(var i = 0; i < this.skills.length; i++) {%>' +
         '<li><a href="#"><%this.skills[i]%></a></li>' +
         '</ul>' +
         '<br/><strong>Total: </strong><%this.skills.length%> main skills' +
         '<%}%>' +
         '<%} else {%>' +
         '<p>none</p>' +
         '<%}%>';

        var result = coreTemplate(template, {
            skills: ["js", "html", "css"],
            showSkills: true
        });

        console.log(result);
