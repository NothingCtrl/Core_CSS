Диалоговые окна

HTML


    <div class="align-center">
        <button class="raised-button bg-green fg-white" onclick="coreDialog.open('#dialog1')">Predefined</button>
        <button class="raised-button bg-orange fg-white" onclick="openCustom()">$.Dialog</button>
    </div>

    <div id="dialog1" data-role="dialog" data-close-button="true" data-overlay="true">
        <div class="dialog-title">
            Use Google's location service?
        </div>
        <div class="dialog-content">
            Bassus abactors ducunt ad triticum. A fraternal form of manifestation is the bliss.
        </div>
        <div class="dialog-actions">
            <button class="flat-button" onclick="coreDialog.close('#dialog1')">disagree</button>
            <button class="flat-button" onclick="coreDialog.close('#dialog1')">agree</button>
        </div>
    </div>

JS


    function openCustom(){
        $.Dialog({
            title: "Dialog title",
            content: "This is a dialog content",
            actions: [
                {
                    title: "Ok",
                    onclick: function(el){
                        alert("You pressed OK button");
                    }
                },
                {
                    title: "Cancel",
                    cls: "js-dialog-close"
                }
            ],
            options: {
                type: "warning"
            }
        });
    }
